# add indicators as needed for each stock in list
# for loop only for dict, each day inside DF is vectorized ops so faster
# watch out for look ahead issue when calculating new series like mean
smadays=30
mmadays=130
lmadays=300
for key, value in stock_dict.iteritems():
    # calculate the 30/130/300 days simple moving average
    stock_dict[key]['sma']=stock_dict[key].rolling(window=smadays)['close'].mean()
    stock_dict[key]['mma']=stock_dict[key].rolling(window=mmadays)['close'].mean()
    stock_dict[key]['lma']=stock_dict[key].rolling(window=lmadays)['close'].mean()
    # calculate 52w high * 20% drop
    stock_dict[key]['h52wstop'] = stock_dict[key].rolling(window=250)['close'].max()*.85
    stock_dict[key]['h52wmin'] = stock_dict[key].rolling(window=250)['close'].max()*.95
    # calculate 20,60,120,180,240 days price change
    #stock_dict[key]['5dchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+5)) -1
    #stock_dict[key]['10dchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+10)) -1
    #stock_dict[key]['1mchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+20)) -1
    #stock_dict[key]['3mchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+60)) -1
    #stock_dict[key]['6mchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+120)) -1
    #stock_dict[key]['12mchg'] = (stock_dict[key]['close'] / stock_dict[key]['close'].shift(+240)) -1
    
    conditions = [((stock_dict[key])['close'] >= (stock_dict[key])['sma']) & 
                  ((stock_dict[key])['sma'] >= (stock_dict[key])['mma']) & 
                  ((stock_dict[key])['mma'] >= (stock_dict[key])['lma']) &
                  ((stock_dict[key])['close'] > (stock_dict[key])['h52wmin']),
                  ((stock_dict[key])['close'] < (stock_dict[key])['h52wstop']),
                  ((stock_dict[key])['close'] < (stock_dict[key])['sma']) & 
                  ((stock_dict[key])['sma'] < (stock_dict[key])['mma'])]

    choices = ["BUY","SELL","SELL"]
    # based on conditions and choices given above, decide buy/hold/sell for each stock
    (stock_dict[key])['bs'] = np.select(conditions, choices, default="HOLD")
    #
    # drop unwanted data to speed up multiprocessing mem copy
    stock_dict[key].drop(['h52wmin'], axis=1, inplace=True)
    stock_dict[key].drop(['sma'], axis=1, inplace=True)
    stock_dict[key].drop(['mma'], axis=1, inplace=True)
    # create individual dfs for multiprocessing
    globals()[key]=stock_dict[key].copy(deep=True)
